EZ Button

tag: button

directive: ez-btn

attributes: look, label, icon

look: primary, info, success, warning, danger, lg, sm, block

label: string

icon: fontawesome icon string

EZ Check

tag: div

directive: ez-check

attributes: items, mode, name, look

items: string comma separated, string array data binding

mode: checkbox(default), radio

name: string, if left empty it defaults to [mode_randomnumber: radio_123456]

look: primary, info, success, warning, danger, lg, sm, block
